\begin{algorithm}[H]
    \caption{Algoritmo Exacto para \emph{CMF} - Descriptivo}
    \begin{algorithmic}[1]
        \Require\Statex
            \begin{itemize}
                \item Un grafo $G$ de $n$ v\'ertices\footnotemark[1] y $m$ aristas.

                \item Una funci\'on $candidatos(K)$, que dado un conjunto de v\'ertices
                    $K$, devuelve el conjunto de v\'ertices de $G$ que son adjacentes
                    a todos los elementos de $K$, cuyo grado es al menos $2|K|$ y tales
                    que su n\'umero de v\'ertice sea mayor al mayor de todos los elementos
                    de $K$.

                \item Una funci\'on $\delta(K)$, que dada una \emph{clique} $K$\footnotemark[2]
                    en $G$, calcula el cardinal de su frontera\footnotemark[3].

            \end{itemize}\Statex
        \Ensure Una \emph{clique} $K$ de $G$ con una frontera 
            $\delta(K)$ de m\'axima cardinalidad.

        \Statex
        \State $K \gets \emptyset$, $K' \gets \emptyset$
        \State $MaximaFrontera \gets 0$
        \For{$nodo$}{$1$}{$n$}
            \State $K' \gets \{nodo\}$
            \State $YaProcesados(K')$\footnotemark[4]$ \gets \emptyset$
            \NoEndIf{$MaximaFrontera < \delta(K')$}
                \State $K \gets K'$

            \While{$K' \neq \emptyset$}
                \If{$candidatos(K') \setminus YaProcesados(K') \neq \emptyset$}
                    \State Sea $nodo' \in candidatos(K') \setminus YaProcesados(K')$
                    \State $K' \gets K' \cup \{nodo'\}$
                    \NoEndIf{$YaProcesados(K' \setminus \{nodo'\}) = \emptyset$}
                        \State $YaProcesados(K') \gets \emptyset$ \footnotemark[5]

                    \NoEndIf{$MaximaFrontera < \delta(K')$}
                        \State $K \gets K'$

                \Else \Comment{Ac\'a est\'a el \emph{Backtracking}}
                    \State $nodoProcesado \gets ultimo(K')$\footnotemark[6]
                    \State $K' \gets K' \setminus \{nodoProcesado\}$
                    \State $YaProcesados(K') \gets YaProcesados(K') \cup \{nodoProcesado\}$

                \EndIf

            \EndWhile

        \EndFor

        \State \Return{$K$}
    \end{algorithmic}
\end{algorithm}

%Footnotes del pseudoc\'odigo
\footnotetext[1]{Asumimos, sin p\'erdida de generalidad, que los v\'ertices est\'an
    numerados de 1 a $n$.}

\footnotetext[2]{Una \emph{clique} no es otra cosa que un conjunto de v\'ertices, con
    la caracter\'istica de inducir un grafo completo en $G$.}

\footnotetext[3]{Como ya se explic\'o, sea $K$ una \emph{clique} de $G$, entonces:
    $\delta(K) = - |K|(|K|-1) + \sum_{v \in K} d(v)$}

\footnotetext[4]{$YaProcesados(K')$ representar\'a a los v\'ertices de $candidatos(K')$
    que ya fueron tenidos en cuenta para el \emph{clique} $K'$.}

\footnotetext[5]{Inicializo $YaProcesados(K')$ s\'olo si no fue ya inicializado.}

\footnotetext[6]{Donde $ultimo(K')$ es el \'ulitmo v\'ertice agregado a $K'$, y sabemos
    que no se indefine pues la guarda del ciclo \emph{While} nos lo asegura.}
